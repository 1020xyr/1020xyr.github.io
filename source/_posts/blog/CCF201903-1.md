# CCF201903-1

```cpp
#include<stdio.h>
int main(){
	int n;
	int min,max,mid;
	scanf("%d",&n);
	
	int num[n];
	for(int i=0;i<n;i++){
		scanf("%d",&num[i]);
	}
	max=num[0]>=num[n-1]?num[0]:num[n-1];
	min=num[0]>=num[n-1]?num[n-1]:num[0];
	if(n%2!=0){
		mid=num[n/2];
		printf("%d %d %d",max,mid,min);
	}
	else{
		int sum=num[n/2-1]+num[n/2];
		if(sum%2==0)
			printf("%d %d %d",max,sum/2,min);
		else{
			double sum_double=sum;
			printf("%d %.1f %d",max,sum_double/2,min);
		}
	} 
	return 0;
}
```
