# CCF201412-1
```cpp
#include<iostream>
using namespace std;

int* init(int n) {
	int* res = new int[n];
	for (int i = 0;i < n;i++) {
		cin >> res[i];
	}
	return res;
}
void fun(int* data, int n) {
	int tmp[n] = {};
	for (int i = 0;i < n;i++) {
		tmp[data[i]]++;
		cout << tmp[data[i]] << ' ';
	}
}

int main() {
	int n;
	cin >> n;
	int* p = init(n);
	fun(p, n);
	return 0;
}
```
