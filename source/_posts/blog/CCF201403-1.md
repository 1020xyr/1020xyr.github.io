# CCF201403-1
```cpp
#include<iostream>
#include<vector>
using namespace std;
void func(){
	int n;
	int ans = 0;
	cin>>n;
	
	int tmp; 
	vector<int> flag(1001,0);
	for(int i=0;i<n;i++){
		cin>>tmp;
		int abs =  tmp>0 ? tmp : -tmp;
		flag[abs]++;
		if(flag[abs]>1)
			ans++;
	}
	
	cout<<ans;
}
int main(){
	func();
	return 0;
}
```
