# CCF201703-1

```cpp
#include<iostream>
#include<algorithm> 
using namespace std;
bool cmp(int i, int j) {
	return i < j;
}
int func() {
	int n, k;
	cin >> n >> k;
	int a[n];
	for (int i = 0;i < n;i++) {
		cin >> a[i];
	}


	int ans = 0;
	int sum = 0;
	for (int i = 0;i < n;i++) {
		sum += a[i];
		if (sum >= k) {
			ans++;
			sum = 0;
		}
	}
	if (sum != 0) {
		ans++;
	}
	return ans;
}
int main() {
	int ans = func();
	cout << ans;
	return 0;
}
```
