# CCF201909-1

```cpp
#include<iostream>
using namespace std;

void func(){
	int n,m;
	cin>>n>>m;
	int T=0,k,P=-1;
	int sum,num,decrease;
	for(int i=0;i<n;i++){
		cin>>sum;
		decrease = 0;
		for(int j=0;j<m;j++){
			cin>>num;
			decrease -= num;
			sum += num;
		}
		T += sum;
		if(decrease>P){
			P = decrease;
			k = i+1; 
		}
	}
	cout<<T<<" "<<k<<" "<<P;
} 
int main(){
	func();
	return 0;
}
```
