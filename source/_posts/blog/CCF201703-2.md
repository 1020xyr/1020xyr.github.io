# CCF201703-2

```cpp
#include<iostream>
#include<vector>
#include<algorithm>
#include<iterator>
using namespace std;
void out(int& i) {
	cout << i << " ";
}
void out_vector(vector<int> res) {
	for_each(res.begin(), res.end(), out);
}
void func() {
	int n;
	cin >> n;
	vector<int> res(n);
	for (int i = 0;i < n;i++) {
		res[i] = i + 1;
	}
	int m;
	cin >> m;
	for (int i = 0;i < m;i++) {
		int val, dist;
		cin >> val >> dist;
		vector<int>::iterator index = find(res.begin(), res.end(), val);
		res.erase(index);
		vector<int>::iterator tmp = index + dist;
		res.insert(tmp, val);
	}
	out_vector(res);
}
int main() {
	func();
	return 0;
}
```
