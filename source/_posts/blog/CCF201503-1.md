#include<iostream>
using namespace std;
int** init(int line,int row) {
	int** data;
	data = new int* [line];
	for (int i = 0;i < line;i++) {
		data[i] = new int[row];
		for (int j = 0;j < row;j++) {
			cin >> data[i][j];
		}
	}
	return data;
}
int** rotate(int** data, int line, int row) {
	int** res = new int*[row];
	for (int j = 0;j < row;j++) {
		res[row - 1 - j] = new int [line];
		for (int i = 0;i < line;i++) {
			res[row - 1 - j][i] = data[i][j];
		}
	}
	return res;
}
void Print(int** data, int line, int row) {
	for (int j = 0;j < row;j++) {
		for (int i = 0;i < line;i++) {
			cout << data[j][i] << ' ';
		}
		cout << endl;
	}
}
int main() {
	int line, row;
	cin >> line >> row;
	int** tmp = init(line, row);
	int** res = rotate(tmp,line, row);
	Print(res, line, row);
}
